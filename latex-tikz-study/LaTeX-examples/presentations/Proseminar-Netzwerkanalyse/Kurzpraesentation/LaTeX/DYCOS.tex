\begin{frame}{DYCOS}
    DYCOS zeichnet sich aus durch:
    \begin{itemize}
        \item \textbf{Geschwindigkeit}: < $10\si{\second}$ für
            \begin{itemize}
                \item $806\,635$ Knoten ($18\,999$ mit Labels)
                \item $4\,414\,135$ Kanten
            \end{itemize}
        \item \textbf{Genauigkeit}: > $65\%$
    \end{itemize}
\end{frame}
