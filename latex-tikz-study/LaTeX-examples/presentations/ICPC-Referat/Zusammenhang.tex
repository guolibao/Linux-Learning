	\begin{block}{Streng zusammenh채ngender Graph}
		Ein streng zusammenh채ngender Graph ist ein gerichteter Graph,
		in dem jeder Knoten von jedem erreichbar ist.
	\end{block}
	\begin{figure}
		\begin{tikzpicture}[->,scale=1.8, auto,swap]
